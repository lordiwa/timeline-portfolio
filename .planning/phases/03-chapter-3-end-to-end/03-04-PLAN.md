```javascript
    { property: 'og:description',  content: () => t('seo.description') },

    { property: 'og:type',         content: 'website' },
    { property: 'og:url',          content: seoConfig.siteUrl },

    { rel: 'alternate', hreflang: 'es',        href: seoConfig.siteUrl },
    { rel: 'alternate', hreflang: 'en',        href: `${seoConfig.siteUrl}?lang=en` },
    { rel: 'alternate', hreflang: 'x-default', href: seoConfig.siteUrl },
  ],
```

NOTA crítica `useHead` API v1.x:
- Para reactive values, usar functions `() => t(key)` o computed refs.
- Static values pueden ser strings simples.
- `script[].textContent` es el campo correcto para JSON-LD (NOT `innerHTML`).
- @unhead/vue auto-handles cleanup en unmount (Vue plugin standard).

<!-- tests/seo/head-tags.test.js — verificar useHead wiring -->

Test approach: @unhead/vue runtime para tests inyecta el `<head>` real en `document.head` cuando el plugin se aplica + use Vue Test Utils mount. Pattern:

```javascript
// tests/seo/head-tags.test.js
import { describe, it, expect, beforeEach } from 'vitest'
import { mount, flushPromises } from '@vue/test-utils'
import { createHead } from '@unhead/vue'
import App from '@/App.vue'
import { createTestI18n } from '../i18n/test-helpers.js'

describe('SEO head tags (SEO-01..04)', () => {
  beforeEach(() => {
    // Clean previous head state from prior test
    document.head.querySelectorAll('[data-test-seo]').forEach(el => el.remove())
  })

  it('T1 — title reactive al locale (SEO-03)', async () => {
    const i18n = createTestI18n({ locale: 'es' })
    const head = createHead()
    const wrapper = mount(App, {
      global: { plugins: [i18n, head], stubs: {/* mock heavy children */} }
    })
    await flushPromises()
    // unhead inyecta async — esperar un tick más
    await new Promise(resolve => setTimeout(resolve, 0))
    expect(document.title).toContain('PENDING — CONTENT-CHECKLIST §4.1' /* es value */)
    // O si Rafael llenó §4.1 con valor real: 'Rafael Matovelle — Portfolio' or similar
    // Multiplexar: mutar locale='en', flushPromises, assert title cambia
    i18n.global.locale.value = 'en'
    await flushPromises()
    await new Promise(resolve => setTimeout(resolve, 0))
    expect(document.title).toContain('PENDING — CONTENT-CHECKLIST §4.1' /* en value, mismo placeholder o real EN */)
  })

  it('T2 — meta description, og:title, og:description reactive (SEO-01 + SEO-03)', async () => {
    const i18n = createTestI18n({ locale: 'es' })
    const head = createHead()
    mount(App, { global: { plugins: [i18n, head] } })
    await flushPromises()
    await new Promise(resolve => setTimeout(resolve, 0))
    const desc = document.querySelector('meta[name="description"]')
    const ogTitle = document.querySelector('meta[property="og:title"]')
    const ogDesc = document.querySelector('meta[property="og:description"]')
    expect(desc).not.toBeNull()
    expect(ogTitle).not.toBeNull()
    expect(ogDesc).not.toBeNull()
  })

  it('T3 — og:image, og:type, og:url static (SEO-01)', async () => {
    const head = createHead()
    mount(App, { global: { plugins: [createTestI18n(), head] } })
    await flushPromises()
    const ogImage = document.querySelector('meta[property="og:image"]')
    const ogType = document.querySelector('meta[property="og:type"]')
    const ogUrl = document.querySelector('meta[property="og:url"]')
    expect(ogImage.getAttribute('content')).toContain('/og-image.png')
    expect(ogType.getAttribute('content')).toBe('website')
    expect(ogUrl.getAttribute('content')).toMatch(/^https?:\/\//)
  })

  it('T4 — hreflang ES / EN / x-default (SEO-04)', async () => {
    const head = createHead()
    mount(App, { global: { plugins: [createTestI18n(), head] } })
    await flushPromises()
    const hreflangs = Array.from(document.querySelectorAll('link[rel="alternate"][hreflang]'))
```
